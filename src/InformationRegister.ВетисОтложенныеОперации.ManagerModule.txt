
Процедура Добавить(пСсылка, пОперация = Неопределено) Экспорт
	
	НаборЗаписей = РегистрыСведений.ВетисОтложенныеОперации.СоздатьНаборЗаписей();
	
	НаборЗаписей.Отбор.Объект.Установить(пСсылка);
	
	Запись = НаборЗаписей.Добавить();
	Запись.Объект = пСсылка;
	Запись.Операция = ?(пОперация = Неопределено, ПредопределенноеЗначение("Перечисление.ВетисОтложенныеОперации.ПоУмолчанию"), пОперация);
	
	НаборЗаписей.Записать();
