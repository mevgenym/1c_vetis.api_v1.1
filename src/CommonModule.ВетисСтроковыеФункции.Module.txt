Функция _СтрРазделить(Знач пСтрока, пРазделитель = ",", пВключатьПустые = Истина) Экспорт
	
	_Массив = Новый Массив;
	
	Пока Истина Цикл
		_Позиция = Найти(пСтрока, пРазделитель);
		Если _Позиция = 0 Тогда Прервать; КонецЕсли;
		_Подстрока = Лев(пСтрока, _Позиция - 1);
		Если пВключатьПустые ИЛИ НЕ ПустаяСтрока(_Подстрока) Тогда
			_Массив.Добавить(_Подстрока);
		КонецЕсли;
		пСтрока = Сред(пСтрока, _Позиция + СтрДлина(пРазделитель));
	КонецЦикла;
	
	Если пВключатьПустые ИЛИ НЕ ПустаяСтрока(пСтрока) Тогда
		_Массив.Добавить(пСтрока);
	КонецЕсли;
	
	Возврат _Массив;
