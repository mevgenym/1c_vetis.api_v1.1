Функция ТекущийПользователь() Экспорт
	
	_ссылка = Ветис.Переменные_Получить("ВетисПользователь");
	
	Если ЗначениеЗаполнено(_ссылка) Тогда
		Возврат _ссылка;
	КонецЕсли;
	
	_имя = ПользователиИнформационнойБазы.ТекущийПользователь().Имя;
	
	Если НЕ ЗначениеЗаполнено(_имя) Тогда
		_имя = "Администратор (ветис)";
	КонецЕсли;
	
	_ссылка = Справочники.Пользователи.НайтиПоКоду(_имя);
	
	Если НЕ ЗначениеЗаполнено(_ссылка) Тогда
		_элемент = Справочники.Пользователи.СоздатьЭлемент();
		_элемент.Код = _имя;
		_элемент.Наименование = _имя;
		_элемент.Записать();
		_ссылка = _элемент.Ссылка;
	КонецЕсли;
	
	Если НЕ Ветис.Настройки_Интеграция() Тогда
		ПараметрыСеанса.ТекущийПользователь = _ссылка;
	КонецЕсли;
	
	Ветис.Переменные_Установить("ВетисПользователь", _ссылка);
	
	Возврат _ссылка;
	
КонецФункции
