Процедура ПриСозданииНаСервере(Отказ, СтандартнаяОбработка)
	
	Если Параметры.Свойство("Guid") Тогда
		Guid = Параметры.Guid;
	КонецЕсли;
	
	Если Параметры.Свойство("Страна") Тогда
		Страна = Параметры.Страна;
	КонецЕсли;
	
	Если ЗначениеЗаполнено(Guid) И НЕ ЗначениеЗаполнено(Страна) Тогда
		Страна = Ветис.Соответствие_ПолучитьСсылку(Guid, "Справочник.ВетисCountry");
		ЕстьСвязь = ЗначениеЗаполнено(Страна);
	ИначеЕсли НЕ ЗначениеЗаполнено(Guid) И ЗначениеЗаполнено(Страна) Тогда
		Guid = Ветис.Соответствие_ПолучитьGuid(Страна);
		ЕстьСвязь = ЗначениеЗаполнено(Guid);
	Иначе
		ЕстьСвязь = Guid = Ветис.Соответствие_ПолучитьGuid(Страна);
	КонецЕсли;
	
	Элементы.ДекорацияСвязь.Картинка = ?(ЕстьСвязь, БиблиотекаКартинок.ВетисЕстьСвязь, БиблиотекаКартинок.ВетисНетСвязи);
	
	КомандаНайтиПоGuidСервер();
	
	Если НЕ ЗначениеЗаполнено(Страна) И ЗначениеЗаполнено(Наименование) Тогда
		Страна = Справочники.ВетисCountry.НайтиПоНаименованию(Наименование);
	КонецЕсли;

&НаСервере
Процедура КомандаНайтиПоGuidСервер()
	
	Если ЗначениеЗаполнено(Guid) Тогда
		
		_Версия20 = Ветис.Версия_2_0();
		
		_country = ?(_Версия20, ВетисIkarService, ВетисIkarService_2_0).GetCountryByGuid(Guid);
		
		Элементы.ГруппаПредупреждение.Видимость = _country = Неопределено;
		Элементы.ГруппаРеквизиты.Видимость = НЕ _country = Неопределено;
		
	КонецЕсли;
	
	Если НЕ _country = Неопределено Тогда
		Наименование = _country.name;
		
	Иначе
		Наименование = "";
	КонецЕсли;
