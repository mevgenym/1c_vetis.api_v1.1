//Операция внесения изменений в реестр наименований продукции производителя
//Параметры:
// type - bs:RegisterModificationType - Тип операции (CREATE, FIND_OR_CREATE, UPDATE)
// enterprise - ent:Enterprise - Предприятие-производитель
// affectedList - prod:ProductItemList - Входящие записи для операции
//   На данный момент разрешается добавлять только по одной позиции.
//   Используется только при удалении позиции
// resultingList - 	prod:ProductItemList - Получаемые в результате операции записи
//   На данный момент разрешается добавлять только по одной позиции.
//   Используется при добавлении и редактировании записи.
Функция PSLModificationOperation(type, enterprise, affectedList = Неопределено, resultingList = Неопределено) Экспорт
	
	_Объект = Создать("PSLModificationOperation");
	
	_Объект.type = type;
	_Объект.enterprise = enterprise;
	
	Если НЕ affectedList = Неопределено Тогда
		_Объект.affectedList = affectedList;
	КонецЕсли;
	
	Если НЕ resultingList = Неопределено Тогда
		_Объект.resultingList = resultingList;
	КонецЕсли;
	
	Возврат _Объект;
